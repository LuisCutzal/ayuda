        protected System.Web.UI.WebControls.Button btnUpload;
        protected System.Web.UI.WebControls.Label lblUploadResult;
        protected System.Web.UI.WebControls.Panel frmConfirmation;
        protected System.Web.UI.HtmlControls.HtmlInputFile oFile;

        protected void Page_Load(object sender, System.EventArgs e)
        {
            
        }
        override protected void OnInit(EventArgs e)
        {
            // CODEGEN: This call is required by the ASP.NET Web Form Designer.
            InitializeComponent();
            base.OnInit(e);
        }

        private void InitializeComponent()
        {
            this.Cargar.Click += new System.EventHandler(this.Cargar_Click);
            this.Load += new System.EventHandler(this.Page_Load);
        }

        protected void Cargar_Click(object sender, System.EventArgs e)
        {
            string strFileName;
            string strFilePath;
            string strFolder;
            strFolder = Server.MapPath("Archivo/./"); //ubicacion del archivo
            // Get the name of the file that is posted.
            strFileName = oFile.PostedFile.FileName;
            strFileName = Path.GetFileName(strFileName);
            if (oFile.Value != "")
            {
                // Create the directory if it does not exist.
                if (!Directory.Exists(strFolder)) //Exists debuelve un direc
                {
                    Directory.CreateDirectory(strFolder);
                }
                // Save the uploaded file to the server.
                strFilePath = strFolder + strFileName;
                if (File.Exists(strFilePath))
                {
                    lblUploadResult.Text = strFileName + " Ya existe.";
                }
                else
                {
                    oFile.PostedFile.SaveAs(strFilePath); //guardar el archivo
                    lblUploadResult.Text = strFileName + " Cargado correctamente.";
                }
            }
            else
            {
                lblUploadResult.Text = "Ningun archivo seleccionado";
            }
            // Display the result of the upload.
            frmConfirmation.Visible = true;
        }

        }
